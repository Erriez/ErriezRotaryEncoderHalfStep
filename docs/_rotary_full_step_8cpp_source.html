124 int RotaryFullStep::read()
125 {
126  int pinState;
127  int rotaryState;
128  unsigned long timeStamp;
129  unsigned long changeTime;
130 
131  // Sample rotary digital pins
132  pinState = (digitalRead(_pin1) << 1) | digitalRead(_pin2);
133 
134  // Determine new state from the pins and state table.
135  _state = pgm_read_byte(&fullStepTable[_state & 0x0f][pinState]);
136 
137  // Check rotary state
138  switch (_state & 0x30) {
139  case DIR_CW:
140  rotaryState = 1;
141  break;
142  case DIR_CCW:
143  rotaryState = -1;
144  break;
145  case DIR_NONE:
146  default:
147  rotaryState = 0;
148  }
149 
150  // Check if rotary changed
151  if (rotaryState != 0) {
152  timeStamp = millis();
153  changeTime = timeStamp - _lastChange;
154  _lastChange = timeStamp;
155 
156  // Check speed change
157  if (changeTime < (_sensitivity / 2)) {
158  rotaryState *= 3;
159  } else if (changeTime < _sensitivity) {
160  rotaryState *= 2;
161  }
162  }
163 
164  return rotaryState;
165 }
