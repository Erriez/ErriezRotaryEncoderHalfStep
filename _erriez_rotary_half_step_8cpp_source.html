115 int RotaryHalfStep::read()
116 {
117  int pinState;
118  int rotaryState;
119  unsigned long timeStamp;
120  unsigned long changeTime;
121 
122  // Sample rotary digital pins
123  pinState = (digitalRead(_pin1) << 1) | digitalRead(_pin2);
124 
125  // Determine new state from the pins and state table.
126  _state = pgm_read_byte(&halfStepTable[_state & 0x0f][pinState]);
127 
128  // Check rotary state
129  switch (_state & 0x30) {
130  case DIR_CW:
131  rotaryState = 1;
132  break;
133  case DIR_CCW:
134  rotaryState = -1;
135  break;
136  case DIR_NONE:
137  default:
138  rotaryState = 0;
139  }
140 
141  // Check if rotary changed
142  if (rotaryState != 0) {
143  timeStamp = millis();
144  changeTime = timeStamp - _lastChange;
145  _lastChange = timeStamp;
146 
147  // Check speed change
148  if (changeTime < (_sensitivity / 2)) {
149  rotaryState *= 3;
150  } else if (changeTime < _sensitivity) {
151  rotaryState *= 2;
152  }
153  }
154 
155  return rotaryState;
156 }
